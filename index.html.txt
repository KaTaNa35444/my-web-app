    <script>
        // Telegram WebApp Setup
        const tg = window.Telegram.WebApp;
        tg.ready();
        tg.expand();

        const user = tg.initDataUnsafe?.user;
        if (user) {
            document.getElementById('user-display-name').innerText = user.first_name || 'Трейдер';
            document.getElementById('user-display-id').innerText = `ID: ${user.id}`;
        }

        const BACKEND_URL = "https://your-backend-service.onrender.com"; // Встав свій Render URL
        let userBalance = 3933.1166; // Початковий баланс для демонстрації

        function updateBalanceDisplay() {
            document.getElementById('total-balance').innerText = `$${userBalance.toFixed(4)}`;
            document.getElementById('assets-contract-val').innerText = `${userBalance.toFixed(4)} USDT`;
        }
        updateBalanceDisplay();

        // Real-Time Prices via Binance WebSocket
        const cryptoSymbols = ['btcusdt', 'ethusdt', 'xrpusdt', 'ltcusdt', 'bchusdt', 'qtumusdt'];
        const cryptoNames = { 'btcusdt': 'BTC', 'ethusdt': 'ETH', 'xrpusdt': 'XRP', 'ltcusdt': 'LTC', 'bchusdt': 'BCH', 'qtumusdt': 'QTUM' };

        const ws = new WebSocket(`wss://stream.binance.com:9443/ws/${cryptoSymbols.map(s => s + '@ticker').join('/')}`);

        ws.onmessage = (event) => {
            const data = JSON.parse(event.data);
            const symbol = data.s.toLowerCase();
            const price = parseFloat(data.c).toFixed(symbol === 'xrpusdt' || symbol === 'qtumusdt' ? 4 : 2);
            const change = parseFloat(data.P).toFixed(2);
            
            updateCryptoRow(symbol, price, change);
        };

        function updateCryptoRow(symbol, price, change) {
            const isPositive = change >= 0;
            const badgeClass = isPositive ? 'bg-green' : 'bg-red';
            const changeSign = isPositive ? '+' : '';
            const coinName = cryptoNames[symbol] || symbol.toUpperCase();

            const html = `
                <div class="crypto-info">
                    <span class="crypto-symbol">${coinName}</span>
                </div>
                <div>
                    <div class="crypto-price">$${price}</div>
                </div>
                <div class="crypto-badge ${badgeClass}">${changeSign}${change}%</div>
            `;

            let rowHome = document.getElementById(`coin-home-${symbol}`);
            if (!rowHome) {
                rowHome = document.createElement('div');
                rowHome.className = 'crypto-row';
                rowHome.id = `coin-home-${symbol}`;
                document.getElementById('home-crypto-list').appendChild(rowHome);
            }
            rowHome.innerHTML = html;

            let rowMarket = document.getElementById(`coin-market-${symbol}`);
            if (!rowMarket) {
                rowMarket = document.createElement('div');
                rowMarket.className = 'crypto-row';
                rowMarket.id = `coin-market-${symbol}`;
                document.getElementById('market-crypto-list').appendChild(rowMarket);
            }
            rowMarket.innerHTML = html;
        }

        // Navigation Switcher
        function switchTab(tabId, el) {
            document.querySelectorAll('.tab-content').forEach(tab => tab.classList.remove('active'));
            document.querySelectorAll('.nav-item').forEach(item => item.classList.remove('active'));
            
            document.getElementById(tabId).classList.add('active');
            if (el) el.classList.add('active');
        }

        // Modals Logic
        function openModal(id) { document.getElementById(id).style.display = 'flex'; }
        function closeModal(id) { document.getElementById(id).style.display = 'none'; }

        function submitWithdrawal() {
            const addr = document.getElementById('withdraw-addr').value;
            const amount = parseFloat(document.getElementById('withdraw-amount').value);

            if (!addr || isNaN(amount) || amount <= 0) {
                alert('Введіть коректну суму та адрес!');
                return;
            }
            if (amount > userBalance) {
                alert('Недостатньо коштів на балансі!');
                return;
            }

            userBalance -= amount;
            updateBalanceDisplay();
            closeModal('withdrawModal');
            alert(`Заявку на виведення ${amount} USDT створено!`);
        }

            alert(`Ордер "${type}" прийнято в обробку!`);
        }
