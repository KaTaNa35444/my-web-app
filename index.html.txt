    <script>
        const tg = window.Telegram.WebApp;
        tg.expand();

        const user = tg.initDataUnsafe?.user;
        const userId = user ? user.id : 99999;

        // ВСТАВ СЮДИ ПОСИЛАННЯ НА СВІЙ СЕРВЕР, КОЛИ ЗАПУСТИТИ ЙОГО НА RENDER/RAILWAY
        const BACKEND_URL = "https://my-tg-backend.onrender.com";
        async function fetchBalance() {
            try {
                const response = await fetch(`${BACKEND_URL}/api/balance?userId=${userId}`);
                const data = await response.json();
                document.getElementById('balance-value').innerText = `${data.balance.toFixed(2)} USDT`;
            } catch (err) {
                document.getElementById('balance-value').innerText = "0.00 USDT (Тест)";
            }
        }

        new TradingView.widget({
            "width": "100%",
            "height": "100%",
            "symbol": "BINANCE:BTCUSDT",
            "interval": "5",
            "timezone": "Etc/UTC",
            "theme": "dark",
            "style": "1",
            "locale": "uk",
            "enable_publishing": false,
            "hide_side_toolbar": true,
            "container_id": "tradingview_chart"
        });

        fetchBalance();

        document.getElementById('deposit-btn').addEventListener('click', () => {
            tg.showPopup({
                title: 'Поповнення',
                message: 'Переведіть USDT (TRC-20) на гаманець:\n\nТВІЙ_ГАМАНЕЦЬ\n\nБаланс оновиться після підтвердження.',
                buttons: [{type: 'ok', text: 'Зрозуміло'}]
            });
        });
